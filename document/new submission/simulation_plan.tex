\begin{center}
    {\LARGE\bfseries\color{titleblue} Simulation Data-Generating Process}

    \vspace{0.35em}
    {\normalsize A compact DGP for the Monte Carlo design}
\end{center}

{\color{lineblue}\rule{\textwidth}{0.8pt}}

\begin{tcolorbox}[
    colback=softblue,
    colframe=lineblue,
    boxrule=0.7pt,
    arc=2pt,
    left=10pt,right=10pt,top=8pt,bottom=8pt
]
For each Monte Carlo replication, generate an i.i.d.\ sample
\[
(\bm{X}_i, W_i, Y_i), \qquad i=1,\dots,n.
\]
\end{tcolorbox}

\textbf{Treatment assignment.}
\[
W_i \sim \mathrm{Bernoulli}(0.5).
\]

\textbf{Covariates.}
\[
\bm{X}_i \in \mathbb{R}^{50},
\qquad
\bm{X}_i \sim N(\bm{0}, \Sigma),
\qquad
\Sigma_{jk} = \rho^{|j-k|},
\qquad
\rho = 0.3.
\]
Only the first five covariates are active in the DGP; the remaining coordinates
$X_{i6},\dots,X_{i50}$ are noise covariates.

\textbf{Outcome model.}
\begin{align*}
Y_i &\sim \mathrm{Bernoulli}(p_i), \\
p_i &= \Lambda\!\left(b(\bm{X}_i) + (W_i-0.5)\tau(\bm{X}_i)\right), \\
\Lambda(u) &= \frac{1}{1+e^{-u}}.
\end{align*}

\textbf{Baseline component.}
\[
b(\bm{X}_i)
=
\max\{0, X_{i1}+X_{i2}, X_{i3}\}
+\max\{0, X_{i4}+X_{i5}\}.
\]

\textbf{Heterogeneous treatment effect.}
\[
\tau(\bm{X}_i)
=
-0.05
+0.15\left(X_{i1}+\log\!\left(1+e^{X_{i2}}\right)\right).
\]

\textbf{Design goals.} This specification is chosen so that:
\begin{itemize}
    \item the simulation follows a standard heterogeneous-treatment benchmark structure,
    \item the feature space is moderately high-dimensional while the true signal is sparse,
    \item the average treatment effect is near zero or slightly negative,
    \item treatment effect heterogeneity is driven primarily by $X_{i1}$ and $X_{i2}$,
    \item the remaining covariates act as irrelevant noise variables.
\end{itemize}
